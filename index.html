firebase.initializeApp(firebaseConfig); const auth = firebase.auth(); const db = firebase.firestore(); const storage = firebase.storage(); // AUTH function registrar() { let email = emailInput(); let pass = passInput(); auth.createUserWithEmailAndPassword(email, pass) .then(()=> alert("Registrado")) .catch(e=> alert(e.message)); } function login() { let email = emailInput(); let pass = passInput(); auth.signInWithEmailAndPassword(email, pass) .then(()=> mostrarApp()) .catch(e=> alert(e.message)); } function emailInput() { return document.getElementById("email").value; } function passInput() { return document.getElementById("password").value; } function mostrarApp() { document.getElementById("auth").style.display = "none"; document.getElementById("app").style.display = "block"; cargarMaterias(); } // MATERIAS function agregarMateria() { let nombre = document.getElementById("materia").value; db.collection("materias").add({ nombre: nombre, user: auth.currentUser.uid }); document.getElementById("materia").value = ""; } function cargarMaterias() { db.collection("materias") .where("user","==",auth.currentUser.uid) .onSnapshot(snapshot => { let html = ""; snapshot.forEach(doc => { let m = doc.data(); html += `

${m.nombre}

`; }); document.getElementById("lista").innerHTML = html; }); } // ELIMINAR function eliminar(id) { db.collection("materias").doc(id).delete(); } // SUBIR ARCHIVO REAL function subirArchivo(e, id) { let archivo = e.target.files[0]; let ref = storage.ref("archivos/" + id + "/" + archivo.name); ref.put(archivo) .then(()=> alert("Archivo subido")); }
